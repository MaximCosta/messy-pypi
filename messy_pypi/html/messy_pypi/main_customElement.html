from __future__ import annotations

from typing import Iterator


class List(list):
    def __init__(cls, *args, **kwargs):
        super().__init__(*args, **kwargs)

    def prepend(cls, value) -> None:
        cls.insert(0, value)

    def rmAllBeside(cls, elt: any = None, index: int = None):
        if index:
            del cls[index + 1:]
        if elt:
            del cls[cls.find(elt) + 1:]

    def rmAllBehind(cls, elt: any = None, index: int = None):
        if index:
            del cls[:index]
        if elt:
            del cls[:cls.find(elt)]

    def rmMAll(cls, elts: list[any]) -> None:
        for elt in elts:
            cls.rmAll(elt)

    def rmAll(cls, elt: any) -> None:
        while elt in cls:
            cls.remove(elt)

    def rm(cls, elt: any) -> None:
        if elt in cls:
            cls.remove(elt)

    def indexExist(cls, index: int) -> bool:
        if 0 <= index < len(cls):
            return True
        return False

    def replace(cls, start: any, end: any) -> None:
        if start in cls:
            cls[cls.index(start)] = end

    def replaceAll(cls, start: any, end: any, maxreplace: int = float('inf')) -> None:
        i = 0
        while (start in cls) and i < maxreplace:
            cls[cls.index(start)] = end

    def find(cls, elt: any) -> int:
        if elt in cls:
            return cls.index(elt)
        return None

    def findAll(cls, elt: any) -> list[int]:
        indexl = list()
        for key, val in enumerate(cls):
            if val == elt:
                indexl.append(key)
        return indexl

    def clearNotValue(cls) -> None:
        for k, v in cls.renumerate:
            if not v:
                del cls[k]

    def clearDuplicate(cls) -> None:
        dup = List(set(cls))
        cls.clear()
        cls.extend(dup)

    def showDuplicate(cls) -> list:
        dup = []
        counta = cls.countAll()
        for i in counta:
            if i[1] > 1:
                dup.append(i[0])
        return dup

    def countAll(cls) -> list[tuple[any, int]]:
        counta = []
        unique = list(set(cls))
        for elt in unique:
            counta.append((elt, cls.count(elt)))
        return counta

    def toType(cls) -> None:
        for key, val in enumerate(cls):
            if val.replace('.', '', 1).lstrip('-').isdigit():
                if '.' in val:
                    cls[key] = float(val)
                else:
                    cls[key] = int(val)

    def include(cls, elt: any) -> bool:
        if elt in cls:
            return True
        return False

    def includes(cls, elts: list) -> bool:
        for elt in elts:
            if elt not in cls:
                return False
        return True

    def __copy__(cls) -> List:
        return List(cls)

    def copy(cls) -> List:
        return List(cls)

    @property
    def enumerate(cls) -> enumerate:
        return enumerate(cls)

    @property
    def renumerate(cls) -> Iterator:
        return reversed(list(cls.enumerate))

    @property
    def maxv(cls) -> any:
        return max(cls)

    @property
    def maxi(cls) -> int:
        return cls.index(max(cls))

    @property
    def maxl(cls) -> any:
        l = list(map(lambda x: len(x), cls))
        return cls[l.index(max(l))]

    @property
    def minv(cls) -> any:
        return min(cls)

    @property
    def mini(cls) -> int:
        return cls.index(min(cls))

    @property
    def minl(cls) -> any:
        l = list(map(lambda x: len(x), cls))
        return cls[l.index(min(l))]

    @property
    def length(cls) -> int:
        return len(cls)
